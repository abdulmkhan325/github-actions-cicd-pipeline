# This is a template for staging EC2 instance creation, 
# reliant on main.tf resources and requiring unique identifiers 
# for each instance.

resource "random_id" "server_RANDOMID" {
  keepers = {
    # Generate a new id each time we switch to a new AMI id
    ami_id = "${var.base_ami_id}"
  }
  byte_length = 8
}

resource "aws_instance" "staging_cicd_RANDOMID" {
  # Read the AMI id "through" the random_id resource to ensure that
  # both will change together.
  ami                    = random_id.server_RANDOMID.keepers.ami_id
  instance_type          = "t2.micro"
  vpc_security_group_ids = ["sg-01c54088241abb5c0"]
  key_name               = aws_key_pair.staging_key.key_name

  tags = {
    "Name" = "staging_cicd_RANDOMID"
  }
}

output "staging_dns_RANDOMID" {
  value = aws_instance.staging_cicd_RANDOMID.public_dns
}
